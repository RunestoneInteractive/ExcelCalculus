Item 7.8.1.9.b.
\begin{equation*} DemandPrice(5)=185. \end{equation*}
\begin{equation*} ProdSurplus= \int_0^{5} (185)- (10+q^2)\ dx=\frac{2500}{3}. \end{equation*}
\begin{equation*} ConsSurplus= \int_0^{5} (210-q^2)\ dx=\frac{250}{3}. \end{equation*}
\begin{equation*} TotalSocialGain=ProdSurplus+ConsSurplus=\frac{2750}{3}. \end{equation*}
